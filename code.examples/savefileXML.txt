XML parser written in CFT.
--------------------------

To parse an XML document, use the Parse function. It expects a single string or a list of strings.
To read from a file, just do

	rootNode = XML:Parse(SomeFile.read).first

Note that this returns a list of XML objects, so usually we add a ".first" to get the first (and
usually only) XML object.


To traverse an XML structure:

	someNode.tagName			# String
	someNode.attributes			# Dict

	someNode.attrGet(attrName, default?)	# Get attribute or "x" if undefined
	someNode.attrGet(attrName)		# Get attribute or null if undefined
	someNode.subNodes(tagName?)   	 	# returs list of &XMLNode objects with tag name (or empty list if none)
	someNode.subNode(tagName?)     		# first subnode with given tag (fails if not found, check with .subnodes() first)
	someNode.allSubNodes(tagName?)
	someNode.getText                	# get list of String from content inside node
	someNode.getString("default")   	# get first String content value
	someNode.getContent             	# Get all content lines trimmed and concatenated
	
To create an XML structure from scratch:

	See test function constructedObject
	To see the result from this function, type
	
		constructedObject.PP
	
	To see full details on the XML structure, type
	
		JSON:PP(constructedObject)
		
	The JSON pretty-printer displays the whole dictionary data structure
	
	
Pretty-printing 

	someNode.PP           # returns List of strings properly formatted
	
Export as string

	someNode.export       # return single string for XML structure
	
XML Namespaces support

	ProcessNameSpaces(node) 
		
	This function parses "xmlns" and "xmlns:xxx" properties for all XML nodes,
	creating NamespaceInfo objects which are associated with each XML node
	in the document tree. These are used to rewrite the tags of all XML nodes
	to the following format:
	
	{http://name.space/something}tag
	
	This is no longer valid for XML export, but it makes it possible to do correct
	traversal of name spaced content, regardless of prefixes used. 
	
	See also the SOAP script Schema and WsdlDocument classes for examples of
	such traversal.

	# WS = 1
	root.sub(" ^n^r^t".unEsc).setIsToken(1) => ws # whitespace
	ws.sub(" ^n^r^t".unEsc,ws)  # loop back

	# Specials = 2
	List("<",">","</","/>","=",":")->t root.addToken(t).setIsToken(2) |

	# Numbers = 3

	root.sub(digits+"-"+"+").setIsToken(3) => integer
	integer.sub(digits,integer) # loop back
	integer.sub(".").sub(digits).setIsToken(3) => float 
	float.sub(digits,float) # loop back


	# Identifiers = 4
	identFirstChars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_"
	identInnerChars = identFirstChars + "0123456789:.-"

	root.sub(identFirstChars).setIsToken(4) => ident
	ident.sub(identInnerChars) => insideIdent 
	insideIdent.sub(identInnerChars, insideIdent) # loop back
	insideIdent.setIsToken(4)

	#insideIdent.sub(":") => colon
	#colon.sub(identFirstChars, ident)  # back to main ident parser
		# (implemented in parser, not tokenizer)

	#insideIdent.sub(".") => dot
	#dot.sub(identFirstChars, ident) # back to main ident parser
		# (implemented in parser, not tokenizer)

	# Strings = 5
	List('"',"'")->c 
		root.sub(c) => insideString
		insideString.setDefault(insideString)
		insideString.sub("\").setDefault(insideString)
		insideString.sub(c).setIsToken(5)
	|

	# Processing instructions <? ... ?> 

	root.addToken("<?").setDefault => x
	x.setDefault(x)
	x.sub("?").setDefault(x)
	x.addToken("?>").setIsToken(6)  # terminates processing instruction, 
    
	# Comments <!-- ... -->

	root.addToken("<!--").setDefault => x
	x.setDefault(x)

	# prevent error on partial end-marker
	x.addToken("-").setDefault(x)
	x.addToken("--").setDefault(x)  
	x.addToken("-->").setIsToken(7)  # terminates comment

	# CDATA = 11

	root.addToken("<[CDATA[") => cdata
	cdata.setDefault(cdata)
	cdata.sub("]").setDefault(cdata)
	cdata.addToken("]]>").setIsToken(11)  # CDATA


	# Everything else = 99  
	root.setDefault.setIsToken(99)
	root.addToken(">").setIsToken(99)
	root.addToken("/").setIsToken(99)


	# return value is the root node
	root

# XML node class
# --
    P(1) as String? => tagName
    P(2,Dict) as Dict => attributes
    P(3,List) => content
    
    self.tagName=tagName
    self.attributes=attributes
    self.content=content
    
    # Create sub-node 
    # --
    self.sub=Lambda{
        P(1) as String => tagName
        x=XMLNode(tagName)
        self.content.add(x)
        x
    }

    # Set self.content to empty list, returns self
    # --
    self.clearContent=Lambda{
		self.content=List
		self
	}

    
    # Add text or CDATA, returns self
    # --
    self.addText=Lambda{
        P(1) as String? => text
        if (text != null) {
            if (text.startsWith("<[CDATA[")) {
                text=text.sub(8, text.length-3) # ]]>
            }
            self.content.add(text)
        }
		self
    }


    # Set single text value into self.content
    # --
    self.setText=Lambda{
		P(1) as String? => text
		self.clearContent
		self.addText(text)
		self
	}

    
    # Get attribute or null if undefined
    # --
    self.attrGet=Lambda{
        P(1) as String => attrName
        P(2) as String? => default
        if (self.attributes.has(attrName), self.attributes.get(attrName), default)
    }
    
    # Get list of XMLNode from content
    # --
    self.subNodes=Lambda{
        P(1) as String? => tag
        self.content->x
            x as? &XMLNode => isNode
            assert(isNode)
            assert(tag==null || x.tagName==tag)
            out(x)
    }

    # Get first XMLNode from content
    # --
    self.subNode=Lambda{
        P(1) as String? => tag
	self.subNodes(tag).first
    }
    
    # Get all sub nodes (recursively)
    # --
    self.allSubNodes=Lambda{
    	self.subNodes->node
    		out(node)
    		node.allSubNodes->x
    			out(x)
    }
    
    # Filter allSubNodes on tag name
    # --
    self.allSubNodesByTag = Lambda{
    	P(1) => tag
    	self.allSubNodes->n
    		assert(n.tagName==tag)
    		out(n)
    }

    # Get list of String from content
    # --
    self.getText=Lambda{
        self.content->x
            x as? String => isString
            assert(isString)
            out(x)
    }

    # Get single (first) String value
    # --
    self.getString=Lambda{
        P(1,null) => default
        list=self.getText
        if (list.empty, default, list.first)
    }

    # Get all strings from content (trimmed)
    # --
    self.getContent=Lambda{
        P(1,null) => default
        list=self.getText
        if (list.empty, default, list.concat(" ").trim)
    }



    
        
    ## Parse from token stream
    ## -----------------------
    
    self.parse=Lambda{
        P(1) as LexerTokenStream => ts

		ws=Lambda{
			P(1) as LexerTokenStream => ts
			loop
				break(ts.EOF || ts.peekType != 1)
				ts.next
		} 

        # START TAG BEGIN + TAG NAME
        
		ws.call(ts)
        error(!ts.peek=="<",ts.sourceLocation + ": expected '<' starting a tag")
        ts.next

		ws.call(ts)
        error(ts.peekType != 4, ts.sourceLocation + ": expected identifier")
        name=ts.peek
        ts.next
        
        
        #println("STARTING tag " + name)
        self.tagName=name

        
        # TAG ATTRIBUTES
        
        Inner {
            loop
				ws.call(ts)
                break(ts.peek==">" || ts.peek=="/>")
                
                error (ts.peekType != 4 && ts.peekType != 5, 
                    ts.sourceLocation + ": Expected identifier or string attribute name - got " + ts.peek)
                
                attrName=ts.peek
                if (ts.peekType==5) {
                    # Strip quotes
                    attrName=attrName.sub(1,attrName.length-1)
                }
                ts.next

				ws.call(ts)
                error(ts.peek != "=", 
                    ts.sourceLocation + ": expected '=' for attribute " + attrName)

                ts.next

				ws.call(ts)
                error(ts.peekType != 4 && ts.peekType != 5 && ts.peekType() != 3, 
                    ts.sourceLocation + ": Expected identifier, string or number value for attribtute " + attrName + " - got "+ ts.peek)

                attrValue=ts.peek
                if (ts.peekType==5) {
                    # Strip quotes
                    attrValue=attrValue.sub(1,attrValue.length-1)
                } 

                ts.next
                
                self.attributes.set(attrName,attrValue)  
        }
        
        # START TAG END

		ws.call(ts)
        
        hasContent=true
        if (ts.peek==">") {
            ts.next
        } else if (ts.peek=="/>") {
            hasContent=false
            ts.next
        } else {
            error(ts.sourceLocation + ": expected '>' or '/>' closing tag " + name)
        }
        
        
        # CONTENT

        if (hasContent) Inner {
            text=""
            loop
                break(ts.peek=="</")
                if(ts.peek=="<") {
                    self.addText(UnescapeXMLString(text)) 
                    text=""

                    # parse sub-node
                    node=XMLNode
                    node.parse(ts)

                    self.content.add(node)
                } else {
                    # Everything else (text)
                    text=text+ts.peek
                    ts.next
                }
            |
            self.addText(UnescapeXMLString(text))
        }    
        

        # CLOSING TAG
        if (hasContent) {
			ws.call(ts)

            error(ts.peek != "</", ts.sourceLocation + ": expected </ closing " + self.tagName)

            ts.next

			ws.call(ts)
            error(ts.peekType != 4, ts.sourceLocation + ": expected identifier")

            name=ts.peek
            ts.next
        
			ws.call(ts)
            error(self.tagName != name, ts.sourceLocation + ": Expected </" + self.tagName + "> - got </" + name + ">")
            ts.match(">","expected '>' following '</"+name)
            
        }
    

    } # self.parse
    
    
    
    ## Pretty-print XML
    ## ----------------
    self.PP=Lambda{
	P(1,"") as String => indent
	P(2,List) as List => result
        
	attributes=Inner{
		self.attributes.keys->attrName
			out(attrName + '="' + self.attributes.get(attrName) + '"')
		| _.concat(" ")
	}
	if (attributes.length>0) attributes=" "+attributes
        
	if (self.content.length==1 && Sys.getType(self.content.first)=="String") {
		# inline presentation of single string content element
		result.add(indent+"<"+self.tagName + attributes + ">" + self.content.first + "</" + self.tagName + ">")
	} else {
		# indented presentation of multiple or non-string content elements
		result.add(indent+"<"+self.tagName + attributes + if(self.content.length==0,"/>",">"))
		indent2=indent+"^t".unEsc
		if (self.content.length>0) {
			Inner {
				self.content->x
					x as? &XMLNode => isNode
					if (isNode) {
						x.PP(indent2,result)
					} else {
						s=x.trim
						if (s.length>0) result.add(indent2 + Util:EscapeXMLString(s))
					}
			}
			result.add(indent+"</"+self.tagName+">")
		}
	}
        
        result
    }
    
    
    # Export XML structure as single line of text
    # --
    self.export=Lambda{
        self.PP->line
            out(line.trim)
        | _.concat(" ")
    }

2026-03-01 RFO

In an attempt to automate SOAP with CURL + CFT, I needed to understand XML namespaces.
SOAP is defined with wsdl files, and WSDL as well as XSD name spaces employ an
attribute 'targetNamespace' for defined content. 

That attribute is not inherited, and since it applies to SOAP, it is not part of the
XML parser (this file). 

The attribute xmlns identifies default namespace for unqualified tags, and is inherited.

To use custom name spaces, we use xmlns:xxx="http://something/something", and these
are inherited down the XML tree as well, unless redefined. It is not considered good
practice to let xsd:schema depend on inheritance, but that's another matter.

The main function is the ProcessNamespaces, which converts tagName of all XML nodes
to "Clark notation" on format {http://somewhere/something}Customer

It stores a reference namespaceInfo into each node. The NamespaceInfo objects form a
tree where each points to its (lexical) parent, enabling inheritance of xmlns and
namespace prefixes.

# Undefined name space on internal format
# --
	"{}"
/NS_undefined

# NamespaceInfo class
# -------------------
# Keep track of namespace stuff; forms a linked list according to XML nesting, in order
# to look up inherited namespaces
# --
	P(1) as &XMLNode => xmlNode
	P(2) => parent

	self.xmlNode=xmlNode
	self.parent=parent
	
	self.xmlns=if(xmlNode.attributes.has("xmlns")) "{" + xmlNode.attrGet("xmlns") + "}" else NS_undefined
	
		# The xmlns is the default namespace for unqualified XML elements in scope
		# Using NS_undefined to mean undefined
	
	# lookup for namespace prefixes	
	self.dict=Dict
	
	# populate namespace prefix lookup dictionary
	xmlNode.attributes.keys->attr 
		assert(attr.startsWith("xmlns:"))
		prefix=attr.after(":")
		value=xmlNode.attributes.get(attr)
		self.dict.set(prefix, "{" + value + "}") 
	|
	
	# look up xmlns property, up the parent chain, fails with error if not found
	self.getXmlns = Lambda{
		if (self.xmlns != XML:NS_undefined) {
			self.xmlns
		} else if (self.parent != null) {
			self.parent.getXmlns
		} else {
			error("No xmlns property found")
		}
	}
	
	# resolve namespaced name/tag to full format, fails with error if failing
	self.resolveNamespace = Lambda{
		P(1) as String => name
		
		alreadyResolved = name.contains("{")
		error(alreadyResolved,"Node namespace already resolved: " + name)
		
		error(!name.contains(":"), "Value has no name space prefix: " + name)
	
		pre=name.before(":")
		
		if (self.dict.has(pre)) {
			self.dict.get(pre) + name.after(":")
		} else if (self.parent != null) {
			self.parent.resolveNamespace(name)
		} else {
			error("Undefined namespace prefix: " + self.xmlNode.nsPrefix + " => " + name)
		}
	}
		
	# apply xmlns to unqualified tag, or look up namespace prefix
	self.expandXMLTag = Lambda{
		P(1) as String => tag
		
		if (!tag.contains(":")) {
			self.getXmlns + ":" + tag
		} else {
			self.resolveNamespace(tag)
		}
	}
	
	# return list of parent nodes, starting with closest parent, upwards
	self.getParentNodes = Lambda{
		curr=self.parent
		loop
			break(curr==null)
			out(curr.xmlNode)
			curr=curr.parent
	}
	
	# look up xmlNode (self or parent) which has a particular property, return null if not found
	self.findNodeByAttribute = Lambda{
		P(1) as String => attrName
		P(2, true) as boolean => includeSelf
		
		found=null
		if (includeSelf && self.xmlNode.attributes.has(attrName)) found=self
		if (!found) Inner{
			self.getParentNodes->node
				if (node.attributes.has(attrName)) found=node
				break(found)
		}
		found
	}
	
		

/class NamespaceInfo


# Traverse XML node tree and replace tag names with fully qualified names
# using Clark notation: {http://example.com/ns}Customer
#
# Note that this no longer is valid XML, but useful for data mining the structure.
#
# Also stores a reference to NamespaceInfo objects inside each node
# --
	P(1) as &XMLNode => node
	P(2) => parentNsInfo  # null for root node
	
	namespaceInfo=NamespaceInfo(node, parentNsInfo)
	
	if (node.tagName.contains(":")) {
		node.nsPrefix=node.tagName.before(":")
	} else {
		node.nsPrefix=null
	}
	node.tagName=namespaceInfo.expandXMLTag(node.tagName)
	node.namespaceInfo=namespaceInfo
	
	node.subNodes->sub ProcessNamespaces(sub, namespaceInfo)
	
/ProcessNamespaces
