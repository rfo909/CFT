This script allows us to sync a directory tree from one server to another, without
relying on FTP, NFS or drive mounts. In other words: Windows.

The code consists of two components, a web server that we start on the remote
host, and a set of client functions, to invoke the corresponding end points on
the web server.

In order to eliminate character set issues, string parameters in calls to the web
server, as well as string (JSON) responses from the web server are hex encoded (UTF-8).

---

The server has no configuration. It runs on port 2500 and awaits requests. Start
it with the CreateServer function.

The client requires a bit of configuration, passed as parameters to the Client class.
Example:

# Create mirror
# --
    SyncRemote:CreateServer
    
    remoteHost="10.0.0.82"
    remoteDir="xxx"
    localDir=Dir("xxx").create
    
    client=SyncRemote:Client(remoteHost, remoteDir, localDir)
    client.sync()
/CreateMirror

If the job is interrupted, it can be restarted. It will traverse the same remote
directory again, but when it comes to copying files in directory, it checks if
each file already exists, and if it does, gets the hash of the remote file, and
compares it to the file in the local dir. Only when they differ is the file
downloaded, replacing the one with bad hash.

# Initialize web server on port 2500 - for testing
# --

    Std.Web.Server(2500) => server
    if (LogFile.exists) LogFile.delete
    server.setServerLogFile(LogFile)
    
    server.RootContext => Root   # The "/" path
    
    # Fetch information about all files in directory
    Root.Context("files", "application/json")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            
            #print("files.path=" + path)
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            dir=Dir(path)
            
            list=List
            dir.files->f
            	dict=Dict
            	dict.name=f.name
            	dict.length=f.length
            	dict.lastModified=f.lastModified
            	list.add(dict)
            |
            obj=SymDict(list)
            JSON:Export(obj).toHexString("UTF-8")
    	})
    
    
    # Fetch information about all subdirs in directory
    Root.Context("dirs", "application/json")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            dir=Dir(path)
            
            list=List
            dir.dirs->d
            	list.add(d.name)
            |
            obj=SymDict(list)
            JSON:Export(obj).toHexString("UTF-8")
    	})
    

    # Use this to decide if a file has been modified at the server, or that
    # download is correct
    # --
    Root.Context("hash","text/plain")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            
            file=File(path)
            
            # return hash string
	    file.hash.toHexString("UTF-8")
    	})
    
    
    Root.Context("file","application/octet-stream")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            start=requestObj.urlParams.start.parseInt
            count=requestObj.urlParams.count.parseInt
            
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            file=File(path)

            binaryData=file.readBinary(start,count)
            
            
            # return binary object
            binaryData
    	})

	P(2) as Dir => localDir
	P(3) as String => remoteDir
	P(4) as String => fileName
	P(5) as int => length

		# server handles mapping to backslash if Windows

	BLOCKSIZE = 4*1024*1024

	targetFile=localDir.file(fileName)
	if (targetFile.exists) targetFile.delete
	
	readPos=0
	loop
		break(readPos>=length)
		count=BLOCKSIZE
		if (readPos+BLOCKSIZE>=length) {
			count=length-readPos	
		}
		#println("readPos=",readPos,"count=",count)
		
		path="/file?path=" + remotePath.toHexString("UTF-8") + "&start=" + readPos + "&count="+count
		result=Std.WebClient.Get(host,2500,path)
	
		error(Sys.getType(result.data) != "Binary","GetFile: Expected binary object")
		
		targetFile.binaryAppend(result.data)
		
		readPos = readPos + count

# Robust GetDirs
# --
	P(1) as String => remoteHost
	P(2) as String => remotePath
	
	loop
		tryCatch(GetDirs(remoteHost, remotePath)) => result
		break(result.ok)
	|		
	result.result
/DoGetDirs

# Creating a class for the replication process
# --
	P(1) as String => remoteHost
	P(2) as String => remoteDir
	P(3) as Dir => localDir
	
	self.remoteHost=remoteHost
	self.remoteDir=remoteDir
	self.localDir=localDir
	
	self.sync=Lambda{
		self.processDir(self.remoteDir, self.localDir)
	}
	self.processDir=Lambda{
		P(1) as String => fromDir
		P(2) as Dir => toDir
		self.copyDirFiles(fromDir,toDir)
		json=SyncRemote:DoGetDirs(self.remoteHost, fromDir)
		#JSON:PP(json,true)
		json.list->dirName
			self.processDir(fromDir+"/"+dirName, toDir.sub(dirName).create)
		
	}
	self.copyDirFiles=Lambda{
		P(1) as String => fromDir
		P(2) as Dir => toDir
		
		json=SyncRemote:DoGetFiles(self.remoteHost, fromDir)
		#JSON:PP(json,true)
		json.list->entry
			fileName=entry.name
			fileLength=entry.length
			
			println(fromDir+"/"+fileName,fileLength,"bytes")
			
			localFile=toDir.file(fileName)
			skipDownload=false
			
			if (localFile.exists) {
				# Running local and remote hash functions in parallel
				#process=SpawnProcess(SymDict(localfile), localFile.hash)
				
				remoteHash=SyncRemote:DoGetHash(self.remoteHost, fromDir, fileName)
				
				#localHash=process.wait.exitValue
				localHash=localFile.hash
				
				if (localHash==remoteHash) skipDownload=true
			}
			if (skipDownload) {
				println("   Hash OK")
			} else {
				SyncRemote:DoGetFile(self.remoteHost, fromDir, fileName, fileLength, toDir)
			}
	}
		
/class Client
