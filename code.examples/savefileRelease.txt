# Project home directory
    Sys.homeDir
/HomeDir


# Create overview lines from *.md file
# --
    P(1) as File => file

    inCode=false
   
    file.read->line
        if (line.startsWith("```")) {
            inCode=!inCode
        }
        reject(inCode)
        assert(line.startsWith("#"))
        
        line
            .replace("###","       ")
            .replace("##", "  | ")
            .replace("#",  "*")
            =>str
        out(str)
//Overview


# Create summaries for *.md
# --
    Inner{
        HomeDir.sub("doc").files("*.md")->file
            result = file.dir.file(file.name.before(".")+".txt")
            lines=Overview(file)
            result.setWriteLF.create(lines)
    }
/CreateSummaries



CreateSummaries
/cs



# Run git add
# ---

    CreateSummaries
    
    # Run git
    HomeDir.runCapture("git","add",".")
    GitStatus

/Add



# Commit changes
# --
    GitStatus
    Lib:ReadValue ("Commit message?", Sys.version) =>msg
    HomeDir.runCapture("git","commit","-m",msg)
/Commit


# Push latest commits
# --
    HomeDir.runCaptureShow("git","push","origin","master") 
	true
/Push





# Git status
# ..
    HomeDir.runCaptureShow("git","status")
	true
/GitStatus




# Git diff
    HomeDir.runCaptureShow("git","diff")
	true
/Diff
