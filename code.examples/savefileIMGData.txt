# Process image, returns list of test strings (hex-coded inputs + outputs)
# --
	P(1,ExampleFile) as File => file
	P(2,10) as int => numSamples
	
	raster=Std.Raster.load(file)

	if (raster.width < 640 && raster.height < 640) {
		# use as is without scaling
		raster=raster.gray()
		GetRasterData(raster,numSamples)
	} else {
		raster=raster.gray().scaleTo(640,640,true)
		GetRasterData(raster,numSamples)
	}
/ProcessImage

# Get raster data for a single raster object 
# --
	P(1) as Raster => img
	P(2,10) as int => numSamples
	
	patternWidth=16+1  # result column to the right
	patternHeight=16
	Std.Data.for(0,numSamples,1)->i
		x=Std.Math.random(img.width-patternWidth).i
		y=Std.Math.random(img.height-patternHeight).i
		str=ToHex(PickInputs(img,x,y))
			+ ":"
			+ ToHex(PickOutputs(img,x,y))
		out(str)
/GetRasterData

# Convert list of bytes to hexstring
# --
	P(1) as List => data
	data->i
		out(i.hex(2))
	| _.concat
//ToHex
			
	
# Pick 256 inputs from an area 16x16 wide. The two rightmost pixels
# columns make up 128 bits. Returns list of bytes for the grey levels
# --
	P(1) as Raster => img
	P(2) as int => x
	P(3) as int => y
	
	Std.Data.for(0,16,1)->dx
	Std.Data.for(0,16,1)->dy
		out(img.pixel(x+dx,y+dy).r)
/PickInputs	

# Pick 16 output values used for back propagation
# --
	P(1) as Raster => img
	P(2) as int => x
	P(3) as int => y
	
	Std.Data.for(0,16,1) -> i
		out(img.pixel(x+16,y+i).r)
/PickOutputs
