			if (targetFile != null) {
				println(targetFile.path)
				targetFile.create(buffer)
			}
			buffer=List
			targetFile=targetDir.file(line.after(" ").trim+".xml")
		} else if (line.trim=="#>>>") {
			indent = indent + 4
		} else if (line.trim=="#<<<") {
			indent = indent - 4
			if (indent < 0) indent=0
		} else if (line.trim.startsWith("#")) {
			if (debugMode) buffer.add(" ".times(indent)+"<!-- DEBUG: " + line.trim.after("#") + " -->")
		} else if (line.trim=="${VALUE}") {
			lastLine=buffer.last + line.trim
			buffer.removeLast.add(lastLine)
			concat=true
		} else if (concat) {
			lastLine=buffer.last + line.trim
			buffer.removeLast.add(lastLine)
			concat=false
		} else {
			buffer.add(" ".times(indent) + line)
		}
	|
	if (targetFile != null) {
		println(targetFile.path)
		targetFile.create(buffer)
	} 

	if (!targetDir) buffer.print

	self.createMessageTemplate = Lambda{

				restriction.subNodes(SOAP:NS_XSD("enumeration"))->enum

			}

			if (!IsSystemType(base)) {
				LookupType(wsdlDocument,base) => typeObject

				if (typeObject.tagName.endsWith("complexType")) {
					self.createComplexTypeTemplate(gen,wsdlDocument,typeObject)
				} else {
					self.createSimpleTypeTemplate(gen,wsdlDocument,typeObject)
				}
			} else {
				# look up name from parent ComplexType which 
				gen.out("${VALUE}")
			}


			gen.out("#<<<")

# Schema processing, main function is processSchemaNode with takes an XMLNode as parameter
# (Which must have tag SOAP:XSD_SCHEMA)
# --
	self.dataStore=null
	self.schemaNode=null
	self.sourceFile=null

	# This function gets called from the Wsdl parser, and when processing imports (below)
	# --
	self.processSchemaNode = Lambda{
		P(1) as &XMLNode => schemaNode
		P(2) as File => sourceFile  
		P(3) as &DataStore => dataStore
		
		self.dataStore=dataStore
		self.dataStore.addSchema(self)

		error(schemaNode.tagName != SOAP:XSD_SCHEMA, "Not an " + SOAP:XSD_SCHEMA + " node")

		self.schemaNode = schemaNode
		self.sourceFile=sourceFile

		self.processImports(schemaNode)
	}
	
	
	self.resolveNamespace = Lambda{
		P(1) as String => name
		error (name.contains("{"), "Already on Clark format!") 
		error (!name.contains(":"), "No namespace prefix!")

		self.schemaNode.namespaceInfo.resolveNamespace(name)
	}

	self.processImports = Lambda{
		P(1) as &XMLNode => schemaNode

		schemaNode.subNodes(SOAP:XSD_IMPORT)->ximp
			ns=ximp.attrGet("namespace")  # must match targetNamespace of schema! (TODO)
			loc=ximp.attrGet("schemaLocation")
			if (loc==null) {
				println("WARNING: ingoring import without schemaLocation attribute")
			} else {
				importFile=self.sourceFile.dir.file(loc)
				println("Processing " + SOAP:XSD_IMPORT + " file: " + importFile.path)

				schema=SOAP:Schema
				XML:Parse(importFile.read).first => newSchemaNode
				XML:ProcessNamespaces(newSchemaNode, self.schemaNode.namespaceInfo)
				
				schema.processSchemaNode(newSchemaNode, importFile, self.dataStore)

				# add the imported node to the document tree
				schemaNode.addSubNode(newSchemaNode)
			}
	}


	# Just verify there are no loose non-namespaced references
	self.checkConsistency = Lambda{
		self.complexTypes.keys->ctname
			self.complexTypes.get(ctname).sequence->ct  # list of Dict with name, type
				name=ct.name
				type=ct.type
				if (!type.contains(":") && !self.complexTypes.has(type) && !self.simpleTypes.has(type)) {
					error(ctName + "." + name + " : UNDEFINED TYPE: " + type)
				}
	}

# Helper, called from UpdateAttributeValuesNamespaces
# Apply default namespace if no namespace prefix, or if no default, resolve prefix if present
# --
	P(1) as &XMLNode => startNode
	P(2) as boolean => directSubNodes  # or all sub-nodes
	P(3) as String => tag
	P(4) as List => attrNames
	P(5) as String? => defaultNamespace
	P(6,false) as boolean => tolerateAlreadyResolved

	nodeList = if(directSubNodes, startNode.subNodes(tag), startNode.allSubNodesByTag(tag))

	if (nodeList.length==0) {
		println
		println("WARNING: no " + if(directSubNodes, "direct", " (any)") + " subnodes found under " + startNode.tagName)
		println("   For tag " + tag)
	}

	updateCount=0
	nodeCount=0

	nodeList->node
		nodeCount=nodeCount+1
		attrNames->attr
			assert(node.attributes.has(attr))

			value=node.attributes.get(attr)
			isResolved=value.contains("{")

			if (!tolerateAlreadyResolved && isResolved) {
				error("Already resolved: " + node.tagName + " attr=" + attr + " value=" + value)
			}

			if (isResolved) {
				# leave value as is
			} else if (value.contains(":")) {
				value=node.namespaceInfo.resolveNamespace(value)
			} else if (defaultNamespace != null) {
				value="{" + defaultNamespace + "}" + value
			} else {
				error("<" + node.tagName + "> Attribute '" + attr + "' with value '" + value + "' must have namespace prefix")
			}
			# update the attribute
			node.attributes.set(attr,value)
			updateCount=updateCount+1
	|
	if (nodeCount>0 && updateCount==0) {
		Lib:Header("WARNING: no updates across " + nodeCount + " nodes")
		println("  startNode=" + startNode.tagName)
		println("  directSubNodes=" + directSubNodes)
		println("  tag=" + tag)
		println("  attrNames=" + attrNames.concat(", "))
	}

# Update various attribute values with regards to name spaces
# Contains a hard-coded list of changes to do to attribute values
# --
	P(1) as &WsdlDocument => wd
	
	# targetNamespace: Top-level subnodes of certain nodes only
	# ---------------------------------------------------------
	
	GET_SUB=true
	GET_ALL=false

	# schema subnodes
	# ---------------
	wd.rootNode.allSubNodesByTag(NS_XSD("schema"))->schema

		assert(schema.attributes.has("targetNamespace"))

		defaultNamespace=schema.attrGet("targetNamespace")  
		acceptResolved=false

		NSResolveAttributes(schema, GET_SUB, NS_XSD("element"), List("name","value","type"), defaultNamespace, acceptResolved)
		NSResolveAttributes(schema, GET_SUB, NS_XSD("complexType"), List("name","type"), defaultNamespace, acceptResolved)
		NSResolveAttributes(schema, GET_SUB, NS_XSD("simpleType"), List("name"), defaultNamespace, acceptResolved)

		Inner{
			# propagate defaultNamespace for complexType -> all element
			schema.subNodes(NS_XSD("complexType"))->node
				NSResolveAttributes(node, GET_ALL, NS_XSD("element"), List("name","type"), defaultNamespace, acceptResolved)
			|
			# same for simpleType -> all restriction
			schema.subNodes(NS_XSD("simpleType"))->node
				NSResolveAttributes(node, GET_ALL, NS_XSD("restriction"), List("base"), defaultNamespace, acceptResolved)
			|
		}
	|

	# wdsl:definitions subnodes
	# -------------------------
	defs = wd.rootNode

	error(defs.tagName != SOAP:NS_WSDL("definitions"), "Invalid defs element")
	
	defaultNamespace=defs.attrGet("targetNamespace")  # or null
	acceptResolved=false

	NSResolveAttributes(defs, GET_SUB, NS_WSDL("message"), List("name"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_SUB, NS_WSDL("portType"), List("name"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_SUB, NS_WSDL("binding"), List("name"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_SUB, NS_WSDL("service"), List("name"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_SUB, NS_WSDL("element"), List("type"), defaultNamespace, acceptResolved)
	
	# Resolve namespace of attributes with no default namespace (requires ns prefix)
	# ------------------------------------------------------------------------------
	defaultNamespace=null
	acceptResolved=true

	# nodes from complete wdsl-document, by tag 

	NSResolveAttributes(defs, GET_ALL, NS_XSD("element"), List("type","value"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_ALL, NS_WSDL("part"), List("element"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_ALL, NS_WSDL("input"), List("message"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_ALL, NS_WSDL("output"), List("message"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_ALL, NS_WSDL("fault"), List("message"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_ALL, NS_WSDL("port"), List("binding"), defaultNamespace, acceptResolved)
	NSResolveAttributes(defs, GET_ALL, NS_WSDL("binding"), List("type"), defaultNamespace, acceptResolved)

	Lib:Header("Show attributes which both have namespace and don't")

	wd.rootNode.allSubNodes->node
		node.attributes.keys->attr
			key=node.tagName+" " + attr
			value=node.attributes.get(attr)
			if (value.contains("{") || value.contains(":")) {
				withNs.set(key,1)
			} else {
				withoutNs.set(key,1)
			}
	|

	# identify shared keys
	withNs.keys->key
		if (withoutNs.has(key)) out(key)
	| => list
	if (list.length==0) {
		println("(None)")
	} else {
		list.print
	}
/ShowAttributesMissingResolve



# (DEBUG) Show attributes without namespace (simple values)

	P(1) as &WsdlDocument => wd

	Lib:Header("Show attributes with simple values")

	dict=Dict
	wd.rootNode.allSubNodes->node
		node.attributes.keys->attr

				str=node.tagName + " "+ attr + "=" + value
				dict.set(str,1)
			}
	|
	dict.keys.sort->str println(str)
/ShowNonNamespacedAttributes

# Create complete HTTP request for a soap call and call it with CURL:Invoke
# --
	P(1,"http://example.com/service") as String => soapService
	P(2,"http://example.com/service/noop") as String => soapAction
	P(3) as String => body
	P(4) as String? => basicAuth
	P(5, false) => debug

	protocol=soapService.before(":")
	targetHostPort=soapService.after("//").before("/")
	targetHost=targetHostPort.before(":")
	targetService=soapService.after("//").from("/")

	if (debug) {
		println("soapService =",soapService)
		println("targetHostPort=",targetHostPort)
		println("targetHost =",targetHost)
		println("targetService =",targetService)
		println("soapAction =", soapAction)
	}

	headers=Dict
		.set("Content-Type", "text/xml;charset=UTF-8")
		.set("SOAPAction", soapAction)
		.set("Host",targetHost)
		.set("User-Agent","Apache-HttpClient/4.5.5 (Java/12.0.1)")
	
	# Note: Content-Length is handled by curl

	if (basicAuth != null) headers.set("Authorization", "Basic " + basicAuth)

	CURL:Invoke(protocol, targetHostPort, "POST", targetService, headers, body, debug)
/Call
