# WSDL file parser
# --
	self.sourceFile=null
	self.messages=Dict
	self.portTypes=Dict
	self.rootNode=null
	
	self.init = Lambda{
		P(1) as File => f
		P(2) as &DataStore => dataStore

		dataStore.addWsdlDocument(self)
		
		self.sourceFile=f
		XML:Parse(f.read).first => root
		XML:ProcessNamespaces(root)

		if (root.tagName != SOAP:WSDL_DEFINITIONS) {
			error("WSDL root element should be of type " + SOAP:WSDL_DEFINITIONS)
		}

		self.rootNode=root
		self.targetNamespace=root.attrGet("targetNamespace")

		root.subNodes(SOAP:NS_WSDL("types"))->wt
			# inline schema
			Inner {
				wt.subNodes(SOAP:XSD_SCHEMA)->schemaNode
					s=SOAP:Schema
					s.processSchemaNode(schemaNode, f, dataStore)
			}
		|
		
		root.subNodes(SOAP:NS_WSDL("message"))->e
			name=e.attrGet("name")
			parts=List
			Inner{
				e.subNodes(SOAP:NS_WSDL("part"))->p
					part = Dict
						.set("name",p.attrGet("name"))
						.set("element",p.attrGet("element"))
						.set("e", p) # XML element
					parts.add(part)
			}
			self.messages.set(name,SymDict(parts, e)) # store XML element
		|

		root.subNodes(SOAP:NS_WSDL("portType"))->e
			name=e.attrGet("name")
			
			operations=List
			Inner{
				e.subNodes(SOAP:NS_WSDL("operation"))->op
				operation=Dict
					.set("name", op.attrGet("name"))
					.set("e", op)  # XML element
				operations.add(operation)
			}
			self.portTypes.set(name,SymDict(operations, e)) # store XML element
		|
			
		# omitting for now:
		# root.subNodes("wsdl:binding")
		# root.subNodes("wsdl:service")

	}

	self.showMessages = Lambda{
		self.messages.keys->key
			report(key)
			self.messages.get(key).parts->p
				report("",p.name,p.element)
	}

	self.showPortTypes = Lambda{
		self.portTypes.keys->key
			report(key)
			self.portTypes.get(key).operations->op
				report("",op.name)
	}

	self.showEverything = Lambda{
		Lib:Header("xmlns")
		println(self.xmlns)

		Lib:Header("targetNamespace")
		println(self.targetNamespace)

		Lib:Header("Messages")
		Lib:ShowRows(self.showMessages)

		Lib:Header("Port types")
		Lib:ShowRows(self.showPortTypes)
	}

/class WsdlDocument



# Schema processing, main function is processSchemaNode with takes an XMLNode as parameter
# (Which must have tag SOAP:XSD_SCHEMA)
# --
	self.dataStore=null
	self.schemaNode=null
	self.sourceFile=null

	self.simpleTypes=Dict
	self.complexTypes=Dict
	self.globalElements=Dict

	# This function gets called from the Wsdl parser, and when processing imports (below)
	# --
	self.processSchemaNode = Lambda{
		P(1) as &XMLNode => schemaNode
		P(2) as File => sourceFile  
		P(3) as &DataStore => dataStore
		
		self.dataStore=dataStore
		self.dataStore.addSchema(self)

		error(schemaNode.tagName != SOAP:XSD_SCHEMA, "Not an " + SOAP:XSD_SCHEMA + " node")

		self.schemaNode = schemaNode
		self.sourceFile=sourceFile

		self.targetNamespace="{" + schemaNode.attrGet("targetNamespace","") + "}"
		error(self.targetNamespace=="{}", "Attribute targetNamespace missing in " + SOAP:NS_XSD("schema"))

		self.processImports(schemaNode)
		self.addSimpleTypes(schemaNode)
		self.addComplexTypes(schemaNode)
		self.processGlobalElements(schemaNode)
	}
	
	self.applyTargetNamespace = Lambda{
		P(1) as String => name
		if (!name.contains(":")) {
			self.targetNamespace + name
		} else {
			self.resolveNamespace(name)
		}
	}
	
	self.resolveNamespace = Lambda{
		P(1) as String => name
		error (name.contains("{"), "Already on Clark format!") 
		error (!name.contains(":"), "No namespace prefix!")

		self.schemaNode.namespaceInfo.resolveNamespace(name)
	}

	self.addSimpleTypes = Lambda{
		P(1) => schema
		schema.subNodes(SOAP:NS_XSD("simpleType"))->t
			name=t.attrGet("name")
			type="Any"
			if (t.subNodes(SOAP:NS_XSD("restriction"))) Inner {
				t.subNode(SOAP:NS_XSD("restriction")).attrGet("base") => type
			}
			name=self.applyTargetNamespace(name)
			type=self.applyTargetNamespace(type)
			e=t
			self.simpleTypes.set(name,SymDict(type,e))
	}

	self.addComplexTypes = Lambda{
		P(1) => schema
		schema.subNodes(SOAP:NS_XSD("complexType"))->t
			typeName=t.attrGet("name")
			sequence=List

			typeName=self.applyTargetNamespace(typeName)

			if (t.subNodes(SOAP:NS_XSD("sequence"))) Inner{
				t.subNode(SOAP:NS_XSD("sequence")).subNodes(SOAP:NS_XSD("element"))->e
					name=e.attrGet("name")
					type=e.attrGet("type")
					
					name=self.applyTargetNamespace(name)
					type=self.applyTargetNamespace(type)
					
					sequence.add(SymDict(name,type,e))
			}
			e=t
			self.complexTypes.set(typeName,SymDict(sequence,e))
	}


	self.processImports = Lambda{
		P(1) as &XMLNode => schema

		schema.subNodes(SOAP:XSD_IMPORT)->ximp
			ns=ximp.attrGet("namespace")  # must match targetNamespace of schema!
			loc=ximp.attrGet("schemaLocation")
			importFile=self.sourceFile.dir.file(loc)
			println("Processing " + SOAP:XSD_IMPORT + " file: " + importFile.path)
			schema=SOAP:Schema
			XML:Parse(importFile.read).first => schemaNode
			XML:ProcessNamespaces(schemaNode, self.schemaNode.namespaceInfo)
			
			schema.processSchemaNode(schemaNode, importFile, self.dataStore)
	}

	self.processGlobalElements = Lambda{  # global elements not inside types
		P(1) as &XMLNode => schema

		schema.subNodes(SOAP:NS_XSD("element"))->elem
			name=elem.attrGet("name")
			type=elem.attrGet("type")

			name=self.applyTargetNamespace(name)
			type=self.applyTargetNamespace(type)
			
			self.globalElements.set(name,type)
	}

	# Just verify there are no loose non-namespaced references
	self.checkConsistency = Lambda{
		self.complexTypes.keys->ctname
			self.complexTypes.get(ctname).sequence->ct  # list of Dict with name, type
				name=ct.name
				type=ct.type
				if (!type.contains(":") && !self.complexTypes.has(type) && !self.simpleTypes.has(type)) {
					error(ctName + "." + name + " : UNDEFINED TYPE: " + type)
				}
	}

	self.showSimple = Lambda{
		self.simpleTypes.keys->name
			dict=self.simpleTypes.get(name)
			report(name, dict.type)
	}

	self.showComplex = Lambda{
		self.complexTypes.keys->name
			report("")
			report(name)
			report("-".times(name.length))
			sequence=self.complexTypes.get(name).sequence
			sequence->row
				report(row.name,row.type)
	}

	self.showGlobalElements = Lambda{
		self.globalElements.keys->name
			type=self.globalElements.get(name)
			report(name,type)
	}


	self.showEverything = Lambda{
		Lib:Header("targetNamespace")
		println(self.targetNamespace)

		Lib:Header("Simple types")
		Lib:ShowRows(self.showSimple)

		Lib:Header("Complex types")
		Lib:ShowRows(self.showComplex)

		Lib:Header("Global elements")
		Lib:ShowRows(self.showGlobalElements)
	}
/class Schema












# Create SOAP XML call
# --
	P(1,"http://example.com/service") as String => serviceNamespace
	P(2,"noop") as String => action
	P(3,Dict) as Dict => params

	paramList=ActionParameters(params,"ser")
	
	<<< EOF
		<?xml version="1.0" encoding="UTF-8"?>
		<soapenv:Envelope 
			xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" 
			xmlns:ser="<<serviceNamespace>>">
		<soapenv:Header/>
		<soapenv:Body>
			<ser:<<action>>>
<<paramList>>
			</ser:<<action>>>
		</soapenv:Body>
		</soapenv:Envelope>
	>>> EOF
	_.mergeExpr
/CreateSoapRequest


# Create action parameters (helper)
# --
	P(1) as Dict => params
	P(2) as String => ns
	P(3,"^t^t^t^t".unEsc) => indent

	params.keys->key
		value=""+params.get(key)
		tag=ns + ":" + key
		out(indent + "<"+tag+">" + Util:EscapeXMLString(value) + "</"+tag+">")
//ActionParameters


# Create basic authentication string
# --
	P(1) as String => username
	P(2) as String => password

	(username + ":" + password).getBytes("UTF-8").base64
/BasicAuth	


# Create complete HTTP request for a soap call and call it with CURL:Invoke
# --
	P(1,"http://example.com/service") as String => soapService
	P(2,"http://example.com/service/noop") as String => soapAction
	P(3) as String => body
	P(4) as String? => basicAuth
	P(5, false) => debug

	protocol=soapService.before(":")
	targetHostPort=soapService.after("//").before("/")
	targetHost=targetHostPort.before(":")
	targetService=soapService.after("//").from("/")

	if (debug) {
		println("soapService =",soapService)
		println("targetHostPort=",targetHostPort)
		println("targetHost =",targetHost)
		println("targetService =",targetService)
		println("soapAction =", soapAction)
	}

	headers=Dict
		.set("Content-Type", "text/xml;charset=UTF-8")
		.set("SOAPAction", soapAction)
		.set("Host",targetHost)
		.set("User-Agent","Apache-HttpClient/4.5.5 (Java/12.0.1)")
	
	# Note: Content-Length is handled by curl

	if (basicAuth != null) headers.set("Authorization", "Basic " + basicAuth)

	CURL:Invoke(protocol, targetHostPort, "POST", targetService, headers, body, debug)
/Call
