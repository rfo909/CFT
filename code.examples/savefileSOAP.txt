# Namespace prefixes
# --
	P(1) as &XMLNode => node
	dict=Dict

	node.attributes.keys->name 
		assert(name.startsWith("xmlns:"))
		dict.set(name.after(":"), node.attributes.get(name))
	|
	dict
/GetNamespacePrefixes
	

# Schema processing
# --
	
	self.sourceFile=null
	self.simpleTypes=Dict
	self.complexTypes=Dict
	self.includedSchema=List
	self.globalElements=Dict

	self.init = Lambda{
		P(1) as File => f

		XML:Parse(f.read).first => schema  # expects xsd:schema element
		self.sourceFile=f
		self.processSchemaNode(schema)
	}

	self.processSchemaNode = Lambda{
		P(1) as &XMLNode => schema
		P(2, self.sourceFile) as File => sourceFile  

		error(schema.tagName != "xsd:schema", "Not an xsd:schema node")

		self.schema = schema
		self.sourceFile=sourceFile
		error(self.sourceFile==null, "No sourcefile, can not resolve includes")

		self.xmlns=schema.attrGet("xmlns")
		self.nsPrefixes=GetNamespacePrefixes(schema)
		self.targetNamespace=schema.attrGet("targetNamespace")

		self.processImports(schema)
		self.addSimpleTypes(schema)
		self.addComplexTypes(schema)
		self.processGlobalElements(schema)
	}

	self.addSimpleTypes = Lambda{
		P(1) => schema
		schema.subNodes("xsd:simpleType")->t
			name=t.attrGet("name")
			type="Any"
			if (t.subNodes("xsd:restriction")) Inner {
				t.subNode("xsd:restriction").attrGet("base") => type
			}
			self.simpleTypes.set(name,type)
	}

	self.addComplexTypes = Lambda{
		P(1) => schema
		schema.subNodes("xsd:complexType")->t
			typeName=t.attrGet("name")
			sequence=List

			if (t.subNodes("xsd:sequence")) Inner{
				t.subNode("xsd:sequence").subNodes("xsd:element")->e
					name=e.attrGet("name")
					type=e.attrGet("type")
					sequence.add(SymDict(name,type))
			}
			self.complexTypes.set(typeName,sequence)
	}


	self.processImports = Lambda{
		P(1) as &XMLNode => schema

		schema.subNodes("xsd:import")->ximp
			ns=ximp.attrGet("namespace")
			loc=ximp.attrGet("schemaLocation")
			xfile=self.sourceFile.dir.file(loc)
			println("Processing import XSD file: " + xfile.path)
			schema=Schema
			schema.init(xfile)
	}

	self.processGlobalElements = Lambda{  # global elements not inside types
		P(1) as &XMLNode => schema

		schema.subNodes("xsd:complexType")->ctype
			ctype.subNodes("xsd:element")->elem
				name=xelement.attrGet("name")
				type=xelement.attrGet("type")
				self.globalElements.set(name,type)
	}

	self.checkConsistency = Lambda{
		self.complexTypes.keys->ctname
			self.complexTypes.get(ctname)->ct  # list of Dict with name, type
				name=ct.name
				type=ct.type
				if (!type.contains(":") && !self.complexTypes.has(type) && !self.simpleTypes.has(type)) {
					error(ctName + "." + name + " : UNDEFINED TYPE: " + type)
				}
	}

	self.showSimple = Lambda{
		self.simpleTypes.keys->name
			type=self.simpleTypes.get(name)
			report(name, type)
	}

	self.showComplex = Lambda{
		self.complexTypes.keys->name
			report("")
			report(name)
			report("-".times(name.length))
			sequence=self.complexTypes.get(name)
			sequence->row
				report(row.name,row.type)
	}

	self.showGlobalElements = Lambda{
		self.globalElements.keys->name
			type=self.globalElements.get(name)
			report(name,type)
	}

	self.showNsPrefixes = Lambda{
		self.nsPrefixes.keys->prefix
			namespace=self.nsPrefixes.get(prefix)
			report(prefix, namespace)
	}

/class Schema

# WSDL file parser
# --
	self.sourceFile=null
	self.schema=List   # list of XsdSchema objects

	self.init = Lambda{
		P(1) as File => f

		self.sourceFile=f
		XML:Parse(f.read).first => root

		root.subNode("wsdl:types")->wt
			# schema
			Inner {
				wt.subNodes("xsd:schema")->xschema
					s=Schema
					s.processSchemaNode(xschema, f)
					self.schema.add(s)
			}
	}				
					
/class WsdlParser

			<ser:<<action>>>
<<paramList>>

		</soapenv:Envelope>

	_.mergeExpr

# Create complete HTTP request for a soap call and call it with CURL:Invoke
# --
	P(1,"http://example.com/service") as String => soapService
	P(2,"http://example.com/service/noop") as String => soapAction
	P(3) as String => body
	P(4) as String? => basicAuth
	P(5, false) => debug

	protocol=soapService.before(":")
	targetHostPort=soapService.after("//").before("/")
	targetHost=targetHostPort.before(":")
	targetService=soapService.after("//").from("/")

	if (debug) {
		println("soapService =",soapService)
		println("targetHostPort=",targetHostPort)
		println("targetHost =",targetHost)
		println("targetService =",targetService)
		println("soapAction =", soapAction)
	}

	headers=Dict
		.set("Content-Type", "text/xml;charset=UTF-8")
		.set("SOAPAction", soapAction)
		.set("Host",targetHost)
		.set("User-Agent","Apache-HttpClient/4.5.5 (Java/12.0.1)")
	
	# Note: Content-Length is handled by curl

	if (basicAuth != null) headers.set("Authorization", "Basic " + basicAuth)

	CURL:Invoke(protocol, targetHostPort, "POST", targetService, headers, body, debug)
/Call
