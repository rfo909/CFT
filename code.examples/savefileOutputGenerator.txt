The point of this script is to provide a general way of emitting output, where
data traversal in later stages may affect earlier stages.

The main class is the Generator. A generator has an internal list, content, which is
populated by calls to the out() function. Sub-generators are added with addSubGenerator,
which takes two optional parameters, a name, and an output handler.

An output handler is an object that is of type &OutputHandler, and which must
implement two functions: addValue() and getResult(). 

The OutputHandler is stored in a separate field inside the Generator.

We create sub-generators, to handle a particular type of data, and those are
added to the internal content list, which also contains added objects
via out() -- which may often be strings, but needs not be.

When it comes time to produceOutput(), there are two cases:

- if there are no Generator objects in the content, and there is an OutputHandler,
  all the data of the content list is added to the OutputHandler, and then
  it's getResult() is called
  
- If either there are Generator objects in content, or there is no OutputHandler,
  the content list is traversed. All non-Generator objects are passed through,
  while for Generator objects, we call the produceOutput, and merges the result
  into the result list

Names
-----
Generator-objects have the option of being given a name. This is optional, but
necessary for the "data collecting" at later stages to affect "earlier" parts.

When creating sub-generators, they have a parent pointer to where they were
created. 

See test function t2 for its use, where we look up a parent Generator, then
access one of its sub-generators, by name, and add data to it.

OutputHandler magic
-------------------
Applications can create their own OutputHandler classes. The t2 test uses
the ListOutputUnique which is redefined into "as OutputHandler" to be
used in that capability. 

It is a very simple example, in that it collects data and eliminates
duplicates (works for strings).

# Output generation class, in order to "patch" earlier generated content
# Does not need to handle just strings.
# --
	P(1) as String? => rootName
	self.name=rootName
	self.parent=null

	self.content=List
	self.outputHandler=null

	# Add sub-generator (name optional, defaults to ListOutput)
	# --
	self.addSubGenerator = Lambda{
		P(1) as String? => name
		P(2) => outputHandler  # optional
		
		if (outputHandler != null) self.setOutputHandler(outputHandler)

		x=OutputGenerator:Generator
		x.name=name
		x.outputHandler=outputHandler
		x.parent=self

		self.content.add(x)
		x
	}
	
	# short for addSubGenerator
	# --
	self.sub = Lambda{
		P(1) as String? => name
		P(2) => outputHandler
		self.addSubGenerator(name,outputHandler)
	}

	
	
	# Set output handler
	# --
	self.setOutputHandler = Lambda{
		P(1) as &OutputHandler => outputHandler
		
		if (!outputHandler.has("addValue") || !outputHandler.has("getOutput")) {
			error("OutputHandler must have addValue() and getOutput() lambdas")
		}
		self.outputHandler=outputHandler
	}

	# locate sub-generator by name
	# --
	self.getSubGenerator = Lambda{
		P(1) as String => name

		result=null
		self.content->x
			break(result)
			assert({x as &Generator})
			assert(x.name==name)
			result=x
		|
		error(result==null,"No sub-generator '" + name + "'")
		result
	}
	
	self.getParent = Lambda{
		P(1) as String => name

		ptr=self
		loop
			break(ptr==null)
			break(ptr.name==name)
			ptr=ptr.parent
		|
		error(ptr==null,"No parent '" + name + "' found")
		ptr
	}

	self.setOutputHandler = Lambda{
		P(1) as &OutputHandler => handler
		error(!handler.has("getOutput"), "OutputHandler object must have getOutput function")
		self.outputHandler=handler
	}

	# get outputHandler, fails if not defined
	self.out = Lambda{
		P(1) => value
		self.content.add(value)
	}

	self.produceOutput = Lambda{
		hasData=false
		hasSubGenerators=false
		
		self.content->x
			if (Sys.getType(x)=="Dict" && x.getName=="Generator") hasSubGenerators=true else hasData=true
			break(hasData && hasSubGenerators)
		|
		
		if (hasData && !hasSubGenerators && self.outputHandler != null) {
			# if there is only data, and we have an outputHandler, use it
			Inner{ self.content->x self.outputHandler.addValue(x) }
			self.outputHandler.getOutput
		} else {
			# outputHandler will not be used, even if defined, because content has subGenerators
			if(self.outputHandler != null) {
				println
				println("WARNING: OutputGenerator:Generator .produceOutput()")
				println("   Generator with name=" + self.name)
				println("   An OutputHandler has been added, but there are also sub-Generators in content.")
				println("   This means the OutputHandler will not be used on the data.")
				println
			}
			
			self.content->x
				if (Sys.getType(x)=="Dict" && x.getName=="Generator") {
					x.produceOutput->value out(value)
				} else {
					out(x)
				}
		}
	}
		
/class Generator



# Basic output handler - unique values
# --
	self.data=Dict
	
	self.addValue = Lambda{
		P(1) as String => value
		self.data.set(value,1)
		self
	}

	self.getOutput = Lambda{
		self.data.keys
	}
/class ListOutputUnique as OutputHandler
