# Process line
# --
    P(1) => line
    P(2) => first
    
	
    if (line.startsWith(".h")) {
        if (!first) {
            readLine("Enter")=>x error(x != "","Aborting")
        }
        println(Curses:Clear)
        Inner {
            Letters7x5:T9x10(line.sub(2).trim)->line 
                str=line #.replace("#","O")
                println(Curses:TxtGreen + str + Curses:Reset)
			|
            println
        }
    } else {
        println(line)
		if (line.trim != "") Sys.sleep(300)
    }
//ProcessLine

    P(1,List) => lines
    first=true
    lines->line 
        ProcessLine(line, first)
        first=false
    |
    "Done"
