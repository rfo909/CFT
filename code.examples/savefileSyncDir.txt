This script allows us to sync a directory tree from one server to another, without
relying on FTP, NFS or drive mounts. In other words: Windows.

The code consists of two components, a web server that we start on the remote
host, and a client that runs on the host where we replicate to.

In order to eliminate character set issues, string parameters in calls to the web
server, as well as string (JSON) responses from the web server are hex encoded (UTF-8).

# Initialize web server on port 2500 - for testing
# --

    Std.Web.Server(2500) => server
    if (LogFile.exists) LogFile.delete
    server.setServerLogFile(LogFile)
    
    server.RootContext => Root   # The "/" path
    
    # Fetch information about all files in directory
    Root.Context("files", "application/json")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            
            #print("files.path=" + path)
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            dir=Dir(path)
            
            list=List
            dir.files->f
            	dict=Dict
            	dict.name=f.name
            	dict.length=f.length
            	dict.lastModified=f.lastModified
            	list.add(dict)
            |
            obj=SymDict(list)
            JSON:Export(obj).toHexString("UTF-8")
    	})
    
    
    # Fetch information about all subdirs in directory
    Root.Context("dirs", "application/json")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            dir=Dir(path)
            
            list=List
            dir.dirs->d
            	list.add(d.name)
            |
            obj=SymDict(list)
            JSON:Export(obj).toHexString("UTF-8")
    	})
    
    
    Root.Context("file","application/octet-stream")
    	.GET(Lambda{
            P(1) as WebRequest => requestObj
            path=requestObj.urlParams.path.fromHexString("UTF-8")
            start=requestObj.urlParams.start.parseInt
            count=requestObj.urlParams.count.parseInt
            
            if (Sys.isWindows) path=path.replace("/","\") else path=path.replace("\","/")
            file=File(path)

            binaryData=file.readBinary(start,count)
            
            
            # return binary object
            binaryData
    	})

# Fetch file
# --
	P(1) as Dir => localDir
	P(2) as String => remoteDir
	P(3) as String => fileName
	P(4) as int => length
	
	remotePath=remoteDir+"/"+fileName 
		# server handles mapping to backslash if Windows

	BLOCKSIZE = 4*1024*1024

	targetFile=localDir.file(fileName)
	if (targetFile.exists) targetFile.delete
	
	readPos=0
	loop
		break(readPos>=length)
		count=BLOCKSIZE
		if (readPos+BLOCKSIZE>=length) {
			count=length-readPos	
		}
		#println("readPos=",readPos,"count=",count)
		
		path="/file?path=" + remotePath.toHexString("UTF-8") + "&start=" + readPos + "&count="+count
		result=Std.WebClient.Get("localhost",2500,path)
	
		error(Sys.getType(result.data) != "Binary","GetFile: Expected binary object")
		
		targetFile.binaryAppend(result.data)
		
		readPos = readPos + count
//GetFile
