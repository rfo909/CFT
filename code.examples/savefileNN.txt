/Abs

# Compute an error number (0-1000)
# --
	P(1) as List => a
	P(2) as List => b

	sum=0.0
	Std.Data.for(0,a.length,1)->i
		af=a.get(i).f
		bf=b.get(i).f

		sum=sum + (Abs(af-bf))
	|
	sum*100000
/ComputeError

# Test

	brain=Std.Brain(2,1,4,2)
	
	ITER=100000
	MAX_LEARNING_FACTOR = 0.04
	EPOCH=100
		# the number of training samples after which we
		# calculate average error, and if it is better than
		# before, save model

	savefile=Savefile

	errorSum=0
	lowestErrorSum=EPOCH*EPOCH*EPOCH

	Std.Data.for(1,ITER,1)->i
		a=Std.Math.random(1)
		b=Std.Math.random(1)
		if (Abs(a-b) < 0.05) continue  # avoid too close to equal while training
		input=List(a,b)
		
		correct=List(
			if(a>b,1,0),
			if(b>a,1,0)
		)
		output=brain.forwardPass(input)

		learningFactor=MAX_LEARNING_FACTOR

		brain.BP(learningFactor, correct)
		
		calculatedError=ComputeError(output,correct)
		errorSum = errorSum + calculatedError
		if (i % EPOCH==0) {
			if (errorSum < lowestErrorSum) {
				lowestErrorSum=errorSum
				brain.export(savefile)
			}
			println("#"+i + " AvgError="+(errorSum/EPOCH).i+"  Lowest=" + (lowestErrorSum/EPOCH).i + " LF=" + learningFactor.fmt("",".",6))
			errorSum=0
		}
	|

	Lib:Header(savefile.path)

	#brain = Std.Brain(savefile)
	TestCalc(brain)
/Main


Main
