# Test

	brain=Std.Brain(List(2,4,2)) 
	# first input layer is input
	# last input layer is output
	
	# inputs are encoded into input layers as bias, which after summing up 
	# zero inputs becomes the rawSum, then after applying the activation
	# function, becomes the activation of the neuron.

	# The Brain object uses "linear" activation function for input layer, which is just
	# a pass through for rawSum, and sigmoid for the output layer, to normalize output
	# in 0-1 range. The "hidden" layers default to leakyReLU for optimal trainability. 
	
	#brain.errorMeanSquared
	#brain.errorMeanAbsolute
	brain.errorLinear
	#brain.useLeakyReLU(2)  # output layer

	ITER=20000
	MAX_LEARNING_FACTOR = 0.08
	EPOCH=100
		# the number of training samples after which we
		# calculate average error, and if it is better than
		# before, save model

	savefile=Savefile

	errorSum=0
	lowestErrorSum=EPOCH*EPOCH*EPOCH

	Std.Data.for(1,ITER,1)->i
		a=Std.Math.random(1)
		b=Std.Math.random(1)
		if (Abs(a-b) < 0.05) continue  # avoid too close to equal while training
		input=List(a,b)
		
		correct=List(
			if(a>b,1,0),
			if(b>a,1,0)
		)
		brain.clearOutputDeltas
		output=brain.forwardPass(input)
		brain.addOutputDeltas(correct)
		#brain.calculateAverageDeltas(1)

		learningFactor=MAX_LEARNING_FACTOR
		brain.BackPropagate(learningFactor)
		
		calculatedError=ComputeError(output,correct)
		errorSum = errorSum + calculatedError
		if (i % EPOCH==0) {
			if (errorSum < lowestErrorSum) {
				lowestErrorSum=errorSum
				brain.export(savefile)
			}
			println("#"+i + " AvgError="+(errorSum/EPOCH).i+"  Lowest=" + (lowestErrorSum/EPOCH).i + " LF=" + learningFactor.fmt("",".",6))
			errorSum=0
		}
	|

	Lib:Header(savefile.path)

	# test on best version
	brain.import(savefile)
	TestCalc(brain)
