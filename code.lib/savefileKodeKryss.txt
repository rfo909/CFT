		} else {
			parts=str.split("=")
			a=parts.first.trim
			b=parts.last.trim
			
			# remove previous mapping for target character
			codes=DeleteCharLookup(codes,b)

			if (a.?parseInt) {
				codes.set(""+a,b)
			} else {
				# replace character
				Inner{
					codes.keys->key
						if (codes.get(key)==a) codes.set(key,b)
				}
			}
		}
		Db2:Set(Sys.scriptId,"Koder", codes)

# Check if dictionary word follows pattern in coded word
# --
	P(1) as String => codedWord
	P(2) as String => dictionaryWord
	P(3, Dict) as Dict => map
	
	dChars=dictionaryWord.chars
	i=0
	fail=false
	codedWord.chars->c
		break(fail)
		if (map.has(c)) { 
			if (dChars.get(i) != map.get(c)) fail=true
		} else {
			# extend map
			map.set(c,dChars.get(i))
		}
		i=i+1
	|
	if (!fail) {
		# also check that no two keys refer to the same resulting char
		reverseMap=Dict
		Inner{map.keys->key 
			break(fail)
			ch=map.get(key)
			if (reverseMap.has(ch) && reverseMap.get(ch) != key) fail=true else reverseMap.set(ch,key)
		}
	}
	if(fail,null,map)
/CheckWord
